## 闭包

### 封装变量

```js
//示例1，cache暴露在全局
var cache = {}; 
var mult = function(){ 
    var args = Array.prototype.join.call( arguments, ',' ); 
    if ( cache[ args ] ){ 
        return cache[ args ]; 
    } 

    var a = 1; 
    for ( var i = 0, l = arguments.length; i < l; i++ ){ 
        a = a * arguments[i]; 
    } 
    return cache[ args ] = a; 
}; 
console.log ( mult( 1,2,3 ) ); // 6
console.log  ( mult( 1,2,3 ) ); // 6

// 改进写法
var mult = (function(){ 
    var cache = {}; 
    return function(){ 
        var args = Array.prototype.join.call( arguments, ',' ); 
        if ( args in cache ){ 
            return cache[ args ]; 
        } 
        var a = 1; 
        for ( var i = 0, l = arguments.length; i < l; i++ ){ 
            a = a * arguments[i]; 
        } 
        return cache[ args ] = a; 
    } 
})();
// 改进2
var mult = (function(){ 
 var cache = {}; // 缓存起来
 var calculate = function(){ // 封装乘积函数，缓存起来
    var a = 1; 
    for ( var i = 0, l = arguments.length; i < l; i++ ){ 
        a = a * arguments[i]; 
    } 
    return a; 
 }; 
 return function(){ // 返回新函数
    var args = Array.prototype.join.call( arguments, ',' ); 
    if ( args in cache ){ 
        return cache[ args ]; 
    }
    return cache[ args ] = calculate.apply( null, arguments ); 
 } 
})();



```

```js
// img 对象经常用于进行数据上报，一些浏览器下会丢失30%左右；
// 原因：代码执行结束后，img局部变量被销毁，还没有来得及发送http请求。

 var img = new Image(); 
 img.src = src; 
};
```

## 高阶函数

有2个特点：

- 函数作为参数被传递
- 函数作为返回值输出

### 函数作为参数被传递

## 高阶函数实现AOP

- AOP(面向切面编程)的主要作用是把一些跟核心业务逻辑模块无关的功能抽离出来，这些跟业务逻辑无关的功能通常包括日志统计、安全控制、异常处理,把这些功能抽离出来之后，再通过`动态置入`的方式渗入业务逻辑中，这样做的好处是可以保证业务逻辑的纯净和高内聚性，其次有利于AOP模块的复用。

```js
// 经典的before和after方法 
Function.prototype.before = function( beforefn ){ 
 var __self = this; // 保存原函数的引用
 return function(){ // 返回包含了原函数和新函数的代理函数 
    beforefn.apply( this, arguments ); // 前置函数调用
    return __self.apply( this, arguments ); // 自身函数调用并返回
 } 
}; 
Function.prototype.after = function( afterfn ){ 
 var __self = this; // 保存原函数的引用
 return function(){ // 返回包含了原函数和新函数的代理函数 
    var ret = __self.apply( this, arguments ); // 自身函数调用并返回
    afterfn.apply( this, arguments ); // 后置函数调用
    return ret; 
 } 
}; 
var func = function(){ 
 console.log( 2 ); 
}; 
func = func.before(function(){ 
 console.log( 1 ); 
}).after(function(){ 
 console.log( 3 ); 
}); 
func();// 1,2,3

```

### 高阶函数的其他应用

```js
// currying(函数柯里化)，也称部分求值，函数不会先求值，传入参数后先用闭包存起来，真正求值的时候一次完成求值。

//例1：计算每月开销的函数
var monthlyCost = 0; 
var cost = function( money ){ 
 monthlyCost += money; 
}; 
cost( 100 ); // 第1天开销
cost( 200 ); // 第2天开销
cost( 300 ); // 第3天开销
//cost( 700 ); // 第30天开销
console.log( monthlyCost ); // 600

// 改进版本
var cost = (function(){ 
    var args = []; 
    return function(){ 
        if ( arguments.length === 0 ){ //求值
            var money = 0; 
            for ( var i = 0, l = args.length; i < l; i++ ){ 
                money += args[ i ];
            } 
            return money; 
        }else{ 
            // 保存
            [].push.apply( args, arguments ); 
        } 
    } 
})(); 
cost( 100 ); // 第1天开销
cost( 200 ); // 第2天开销
cost( 300 ); // 第3天开销

console.log( cost() ); // 600

// 提炼柯里化函数，有参数存储，无参数就执行
var currying = function( fn ){ 
 var args = []; 
 return function(){ 
    if ( arguments.length === 0 ){ 
        return fn.apply( this, args ); 
    }else{ 
        [].push.apply( args, arguments ); 
        return arguments.callee; 
    } 
 } 
}; 
var cost1 = (function(){ 
 var money = 0; 
 return function(){ 
    for ( var i = 0, l = arguments.length; i < l; i++ ){ 
        money += arguments[ i ]; 
    } 
    return money; 
 } 
})(); 
var cost = currying( cost1 ); // 转化成curry函数
cost( 100 ); // 未真正求值
cost( 200 ); //未真正求值
cost( 300 ); // 未真正求值
console.log( cost() ); //求值600

//  uncurrying 反柯里化，借用函数
Function.prototype.uncurrying = function () { 
 var self = this; // self就是Array.prototype.push
 return function() { 
    var obj = Array.prototype.shift.call( arguments );// 提取上下文
    return self.apply( obj, arguments ); 
 }; 
};
// 另一种实现
Function.prototype.uncurrying = function(){ 
    var self = this; 
    return function(){ 
        return Function.prototype.call.apply( self, arguments ); 
    } 
};

//示例：
var push = Array.prototype.push.uncurrying(); 
(function(){ 
 push( arguments, 4 ); 
 console.log( arguments ); // [1, 2, 3, 4] 
})( 1, 2, 3 );

//
for ( var i = 0, fn, ary = [ 'push', 'shift', 'forEach' ]; fn = ary[ i++ ]; ){ 
 Array[ fn ] = Array.prototype[ fn ].uncurrying(); 
}; 
var obj = { 
 "length": 3, 
 "0": 1, 
 "1": 2, 
 "2": 3 
}; 
Array.push( obj, 4 ); // 向对象中添加一个元素
console.log( obj.length ); // 4
var first = Array.shift( obj ); // 截取第一个元素
console.log( first ); // 1
console.log( obj ); // {0: 2, 1: 3, 2: 4, length: 3} 
Array.forEach( obj, function( i, n ){ 
 console.log( n ); // 0, 1, 2 
});

// 示例2
var apply = Function.prototype.apply.uncurrying(); 
var fn = function( name ){ 
 console.log( this.name ); // "sven" 
 console.log( arguments ); // [1, 2, 3] 
}; 
apply( fn, { name: 'sven' }, [ 1, 2, 3 ] );


```

### 节流函数

- 应用于函数被频繁调用的场景,是一种限制函数被频繁调用的解决方案。
  - window.onresize
  - mousemove
  - 上传进度监听

```js
// 节流函数,
var throttle = function ( fn, interval ) { 
    var __self = fn, // 保存需要被延时执行的函数引用
    timer, // 定时器
    firstTime = true; //是否第一次调用
    return function () { 
        var args = arguments, 
        __me = this; 
        if ( firstTime ) { // 如果第一次调用，不需要延时
            __self.apply(__me, args); 
            return firstTime = false; 
        } 
        if ( timer ) { // 如果定时器还在，上一次未执行完成
            return false;
        } 
        timer = setTimeout(function () { // 延迟一段时间执行
            clearTimeout(timer); 
            timer = null; 
            __self.apply(__me, args); 
        }, interval || 500 ); 
    }; 
}; 
window.onresize = throttle(function(){ 
    console.log( 1 ); 
}, 500 );

```

### 分时函数

- 应用于短时间加载大量数据的场景，是一种分批次初始化数据的解决方案
- 如登录webqq，短时间初始化成千上万个好友数据和聊天记录，浏览器会吃不消而卡死。

```js

var ary = []; 
for ( var i = 1; i <= 1000; i++ ){ 
    ary.push( i ); // 假设 ary 装载了 1000 个好友的数据
}; 
var renderFriendList = function( data ){ 
    // 创建1千个div，并挂载到dom上
    for ( var i = 0, l = data.length; i < l; i++ ){ 
        var div = document.createElement( 'div' ); 
        div.innerHTML = i; 
        document.body.appendChild( div ); 
    } 
}; 
renderFriendList( ary ); //浏览器卡死
// 解决方案就是用分时函数，分批次创建dom节点，1s创建1000个，改成每隔200ms创建8个
// 分时函数
var timeChunk = function( ary, fn, count ){ 
 var obj, 
 t; 
 var len = ary.length; 
 var start = function(){ 
    for ( var i = 0; i < Math.min( count || 1, ary.length ); i++ ){ 
        var obj = ary.shift(); 
        fn( obj ); 
    } 
 }; 
 return function(){ 
    t = setInterval(function(){ 
        if ( ary.length === 0 ){ // 如果全部节点都已经被创建好
            return clearInterval( t ); 
        } 
        start(); 
    }, 200 ); // 分批执行的时间间隔，也可以用参数的形式传入
 }; 
};

// 1000个好友的数据
 var ary = []; 
 for ( var i = 1; i <= 1000; i++ ){ 
    ary.push( i ); 
 }; 
 var renderFriendList = timeChunk( ary, function( n ){ 
    var div = document.createElement( 'div' ); 
    div.innerHTML = n; 
    document.body.appendChild( div ); 
 }, 8 ); 

 renderFriendList();

```

### 惰性加载函数

- 浏览器的差异性，导致编写代码时存在很多嗅探环境的工作。

```js
// 注册事件监听函数，ie与其他浏览器不同。本例存在的问题在于：每次绑定都需要做嗅探工作。
var addEvent = function( elem, type, handler ){ 
 if ( window.addEventListener ){ 
    return elem.addEventListener( type, handler, false );
 } 
 if ( window.attachEvent ){ 
    return elem.attachEvent( 'on' + type, handler ); 
 } 
};
// 改进方案1，从头到尾都用不到window.attachEvent，这个分支是多余的执行
var addEvent = (function(){ 
    if ( window.addEventListener ){ 
        return function( elem, type, handler ){ 
            elem.addEventListener( type, handler, false ); 
        } 
    } 
    if ( window.attachEvent ){ 
        return function( elem, type, handler ){ 
            elem.attachEvent( 'on' + type, handler ); 
        } 
    } 
})();

// 3，惰性函数方案，第一次重写函数，下次进入直接使用
// <html> 
//     <body> 
//         <div id="div1">点击绑定事件</div> 
//         <script> 
        var addEvent = function( elem, type, handler ){ 
            if ( window.addEventListener ){ 
                // 重新定义
                addEvent = function( elem, type, handler ){ 
                    elem.addEventListener( type, handler, false ); 
                } 
            }else if ( window.attachEvent ){ 
                // 重新定义
                addEvent = function( elem, type, handler ){ 
                    elem.attachEvent( 'on' + type, handler ); 
                } 
            } 
            addEvent( elem, type, handler ); 
        };
        var div = document.getElementById( 'div1' ); 
        addEvent( div, 'click', function(){ 
            console.log (1); 
        }); 
        addEvent( div, 'click', function(){ 
            console.log (2); 
        });
```
