# 代码重构

- 在讲设计模式的时候，总是先讲一段反例，再用设计模式重构代码;可以加深对模式的理解
- 模式和重构之间有与生俱来的关系；设计模式的目的之一就是重构代码

## 提炼函数

- 如果在函数中有一段代码可以独立出来，最好把他独立出来。优化的好处
  - 避免出现超大函数
  - 独立函数有助于代码复用
  - 独立函数有助于更容易重写
  - 独立函数有个好的命名，本身就是一种注释

```js

var getUserInfo = function(){ 
 ajax( 'http:// xxx.com/userInfo', function( data ){ 
    console.log( 'userId: ' + data.userId ); 
    console.log( 'userName: ' + data.userName ); 
    console.log( 'nickName: ' + data.nickName ); 
 }); 
};
// 改写后
var getUserInfo = function(){ 
 ajax( 'http:// xxx.com/userInfo', function( data ){ 
    printDetails( data ); 
 }); 
}; 
var printDetails = function( data ){ 
    console.log( 'userId: ' + data.userId ); 
    console.log( 'userName: ' + data.userName ); 
    console.log( 'nickName: ' + data.nickName ); 
};
```

## 把条件分支语句提炼成函数

```js
var getPrice = function( price ){ 
    var date = new Date(); 
    if ( date.getMonth() >= 6 && date.getMonth() <= 9 ){ // 夏天
        return price * 0.8; 
    } 
    return price; 
};
// 改写
var isSummer = function(){ 
    var date = new Date(); 
    return date.getMonth() >= 6 && date.getMonth() <= 9; 
}; 
var getPrice = function( price ){ 
    if ( isSummer() ){ // 夏天

    }
```

```js
    }
```

## 提前让函数退出代替嵌套条件分支

```js
var del = function (obj) {
  var ret;
  if (!obj.isReadOnly) { // 不为只读的才能删除
    if (obj.isFolder) { // 如果是文件夹
      ret = deleteFolder(obj);
    } else if (obj.isFile) { //如果是文件
      ret = deleteFile(obj);
    }
  }
  return ret;
};
// 改写
var del = function (obj) {
  if (obj.isReadOnly) { // 反转if表达式
    return;
  }
  if (obj.isFolder) {
    return deleteFolder(obj);
  }
  if (obj.isFile) {
    return deleteFile(obj);
  }
};
```

## 合理使用链式调用

- 如何链条稳定，后期改动小，可以使用。
- 如果链条可维护性和调试性差，建议改成普通写法。

```js
// 不建议
var User = {
  id: null,
  name: null,
  setId: function (id) {
    this.id = id;
    return this;
  },
  setName: function (name) {
    this.name = name;
    return this;
  }
};
console.log(User.setId(1314).setName('sven'));
// 建议
var user = new User();
user.setId(1314);
user.setName('sven');

```

## 分解大型类

```js
var Spirit = function (name) {//精灵类
  this.name = name;
};
Spirit.prototype.attack = function (type) { // 攻击,可能有很多
  if (type === 'waveBoxing') {
    console.log(this.name + ': 使用波动拳');
  } else if (type === 'whirlKick') {
    console.log(this.name + ': 使用旋风腿');
  }
};
var spirit = new Spirit('RYU');
spirit.attack('waveBoxing'); // RYU: 使用波动拳
spirit.attack('whirlKick'); // RYU: 使用旋风腿

// 改写1，Attack 独立出来
var Attack = function (spirit) {//攻击类
  this.spirit = spirit;
};
Attack.prototype.start = function (type) {
  return this.list[type].call(this);
};
Attack.prototype.list = {
  waveBoxing: function () {
    console.log(this.spirit.name + ': 使用波动拳');
  },
  whirlKick: function () {
    console.log(this.spirit.name + ': 使用旋风腿');
  }
};
var Spirit = function (name) {//精灵类
  this.name = name;
  this.attackObj = new Attack(this);
};
Spirit.prototype.attack = function (type) { // 攻击
  this.attackObj.start(type);
};
var spirit = new Spirit('RYU');
spirit.attack('waveBoxing'); // RYU: 使用波动拳
spirit.attack('whirlKick'); // RYU: 使用旋风腿

```

## 用return退出多层循环

```js
// 1、控制标记flag
var func = function () {
  var flag = false;
  for (var i = 0; i < 10; i++) {
    for (var j = 0; j < 10; j++) {
      if (i * j > 30) {
        flag = true;
        break;
      }
    }
    if (flag === true) {
      break;
    }
  }
};
// 循环标记 outerloop，innerloop
var func = function () {
  outerloop:
  for (var i = 0; i < 10; i++) {
    innerloop:
    for (var j = 0; j < 10; j++) {
      if (i * j > 30) {
        break outerloop;
      }
    }
  }
};
// 直接return
var func = function () {
  for (var i = 0; i < 10; i++) {
    for (var j = 0; j < 10; j++) {
      if (i * j > 30) {
        return;
      }
    }
  }
  console.log(i); // 这句代码没有机会执行
};

// good 
var print = function (i) {
  console.log(i);
};
var func = function () {
  for (var i = 0; i < 10; i++) {
    for (var j = 0; j < 10; j++) {
      if (i * j > 30) {
        return print(i);
      }
    }
  }
};
func();
```
