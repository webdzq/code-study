# 开放-封闭原则(OCP)

- 软件实体（类、函数、模块）等应该是可扩展的，但不可修改
- 在发布-订阅模式、模板方法模式、策略模式、代理模式、职责链模式中都有体现

```js
// 基础实现
window.onload = function () {
  // 500+行代码忽略
  console.log(document.getElementsByTagName('*').length);
};
//新增需求后，after进行扩展
Function.prototype.after = function (afterfn) {
  var __self = this;
  return function () {
    var ret = __self.apply(this, arguments);
    afterfn.apply(this, arguments);
    return ret;
  }
};
window.onload = (window.onload || function () { }).after(function () {
  console.log(document.getElementsByTagName('*').length);
});

```

## 应用技巧

- 用多态消除条件分支
- 放置钩子, 参考11章的hooks
- 使用回调，如ajax等
- 提取不变和变化的部分

```js
var makeSound = function (animal) {
  if (animal instanceof Duck) {
    console.log('嘎嘎');
  } else if (animal instanceof Chicken) {
    console.log('咯咯֬֬֬');
  }
};
var Duck = function () { };
var Chicken = function () { };
makeSound(new Duck()); //嘎嘎
makeSound(new Chicken()); // 咯咯֬֬֬
// 新增一条狗
var makeSound = function (animal) {
  if (animal instanceof Duck) {
    console.log('ً嘎嘎');
  } else if (animal instanceof Chicken) {
    console.log('咯咯֬֬֬֬֬֬');
  } else if (animal instanceof Dog) { // 新增汪汪
    console.log('汪汪');
  }
};
var Dog = function () { };
makeSound(new Dog()); // 加汪汪
// 用多态实现
var makeSound = function (animal) {
  animal.sound();
};
var Duck = function () { };
Duck.prototype.sound = function () {
  console.log('ًً嘎嘎');
};
var Chicken = function () { };
Chicken.prototype.sound = function () {
  console.log('֬֬֬咯咯֬֬֬֬֬֬');
};
makeSound(new Duck()); // ًًً
makeSound(new Chicken()); // ֬֬֬
// 新增汪汪
var Dog = function () { };
Dog.prototype.sound = function () {
  console.log('汪汪');
};
makeSound(new Dog()); // 汪汪

//回调的例子，分离不变与变化的地方
var arrayMap = function (ary, callback) {// 计算规则不同
  var i = 0,
    length = ary.length,
    value,
    ret = [];
  for (; i < length; i++) {
    value = callback(i, ary[i]);
    ret.push(value);
  }
  return ret;
}
var a = arrayMap([1, 2, 3], function (i, n) {
  return n * 2;
});
var b = arrayMap([1, 2, 3], function (i, n) {
  return n * 3;
});
console.log(a); // [ 2, 4, 6 ] 
console.log(b); // [ 3, 6, 9 ]

```
