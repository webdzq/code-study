# 发布-订阅模式

- 也称 观察者模式，定义对象间的一种一对多的依赖关系，当一个对象发现变化时，依赖它的对象都会收到通知
- 也可以用于多对多的场景

## 自定义事件

- 以小红、小龙、小明到售楼处咨询买楼的事情为例，

```js
// 1.简单的实现，问题：订阅消息发送所有人，没有区分特定的人，发生特定消息。
var salesOffices = {}//定义售楼处
salesOffices.clientList = []; // 缓存列表，存放订阅者的回调函数
salesOffices.listen = function( fn ){ // 增加订阅者
    this.clientList.push( fn ); // 订阅的消息添加进缓存列表
}; 
salesOffices.trigger = function(){ // 发布消息
    for( var i = 0, fn; fn = this.clientList[ i++ ]; ){ 
        fn.apply( this, arguments ); // (2) // arguments 是发布消息时带上的参数
    } 
};
//测试
salesOffices.listen( function( price, squareMeter ){ // 小明订阅消息
 console.log( '价格= ' + price ); 
 console.log( 'squareMeter= ' + squareMeter ); 
}); 
salesOffices.listen( function( price, squareMeter ){ // 小红订阅消息
 console.log( '价格= ' + price );
 console.log( 'squareMeter= ' + squareMeter ); 
}); 
salesOffices.trigger( 2000000, 88 ); // 输出：200 万，88 平方米，发送给小明和小红
salesOffices.trigger( 3000000, 110 ); // 输出：300 万，110 平方米，发送给小明和小红

// 2,升级版，上边的实现没有区分订阅类型，都是全量发送
var salesOffices = {}; // 定义售楼处
salesOffices.clientList = {}; // 缓存列表，存放订阅者的回调函数，使用`对象存储`
salesOffices.listen = function( key, fn ){  // 新增key标识
    if ( !this.clientList[ key ] ){ // 如果还没有订阅过此类消息，给该类消息创建一个缓存列表
        this.clientList[ key ] = []; 
    } 
    this.clientList[ key ].push( fn ); // 订阅的消息添加进消息缓存列表
}; 
salesOffices.trigger = function(){ // 发布消息
    var key = Array.prototype.shift.call( arguments ), // 取出消息类型
    fns = this.clientList[ key ]; // 取出该消息对应的回调函数集合
    if ( !fns || fns.length === 0 ){ // 如果没有订阅该消息，则返回
        return false; 
    } 
    for( var i = 0, fn; fn = fns[ i++ ]; ){ 
        fn.apply( this, arguments ); // (2) // arguments 是发布消息时附送的参数
    } 
}; 
//测试
salesOffices.listen( 'squareMeter88', function( price ){ // 小明订阅 88 平方米房子的消息
 console.log( '价格= ' + price ); // 输出： 2000000 
}); 
salesOffices.listen( 'squareMeter110', function( price ){ // 小红订阅 110 平方米房子的消息
 console.log( '价格= ' + price ); // 输出： 3000000 
}); 
salesOffices.trigger( 'squareMeter88', 2000000 ); // 发布 88 平方米房子的价格
salesOffices.trigger( 'squareMeter110', 3000000 ); // 发布 110 平方米房子的价格
```

```js
 login.trigger( 'loginSucc', data); // 发布登录成功的消息
```

## 全局的发布订阅对象

- 如果小明订阅不同房产公司的88、100、300的房子信息
- 小明不需要到售楼处去登记
- 多家房产公司售楼部发布卖楼信息
- 多对多

```js
// 随着业务复杂度增加，进一步升级，Event作为中介公司，剥离买房者和买房者之间的关联。提供发布和订阅功能。
var Event = (function () {
  var clientList = {}, // 订阅列表
    listen,
    trigger,
    remove;
  listen = function (key, fn) {// 订阅
    if (!clientList[key]) {
      clientList[key] = [];// 核心代码，key的定义，及二维数组的产生
    }
    clientList[key].push(fn);
  };
  trigger = function () { // 发布
    var key = Array.prototype.shift.call(arguments),
      fns = clientList[key];
    if (!fns || fns.length === 0) {
      return false;
    }
    for (var i = 0, fn; fn = fns[i++];) {
      fn.apply(this, arguments);
    }
  };
  remove = function (key, fn) { // 移除
    var fns = clientList[key];
    if (!fns) {
      return false;
    }
    if (!fn) {
      fns && (fns.length = 0);
    } else {
      for (var l = fns.length - 1; l >= 0; l--) {
        var _fn = fns[l];
        if (_fn === fn) {
          fns.splice(l, 1);
        }
      }
    }
  };
  return {
    listen: listen,
    trigger: trigger,
    remove: remove
  }
})();
Event.listen('squareMeter88', function (price) { // 小红订阅消息
  console.log('价格=' + price); // 2000000' 
});
Event.trigger('squareMeter88', 2000000); //售楼部发布消息
```

## 新的需求

- 模块间通信,搞不清楚事件从哪里来。
- 全局使用后，事件名冲突问题
- 可以先发布后使用吗?如：qq离线消息接收，企业微信离线消息接收等

```js

var Event = (function () {
  var global = this,
    Event,
    _default = 'default';
  Event = function () {
    var _listen,
      _trigger,
      _remove,
      _slice = Array.prototype.slice,
      _shift = Array.prototype.shift,
      _unshift = Array.prototype.unshift,
      namespaceCache = {},
      _create,
      find,
      each = function (ary, fn) {// 遍历并执行，ary=缓存函数列表
        var ret;
        for (var i = 0, l = ary.length; i < l; i++) {
          var n = ary[i];
          ret = fn.call(n, i, n);
        }
        return ret;
      };
    _listen = function (key, fn, cache) {// 订阅
      if (!cache[key]) {
        cache[key] = [];
      }
      cache[key].push(fn);
    };
    _remove = function (key, cache, fn) {//移除
      if (cache[key]) {
        if (fn) {
          for (var i = cache[key].length; i >= 0; i--) {
            if (cache[key][i] === fn) {
              cache[key].splice(i, 1);
            }
          }
        } else {
          cache[key] = [];
        }
      }
    };
    _trigger = function () {// 发布,stack是订阅函数存储列表
      var cache = _shift.call(arguments),
        key = _shift.call(arguments),
        args = arguments,
        _self = this,
        ret,
        stack = cache[key];
      if (!stack || !stack.length) {
        return;
      }
      return each(stack, function () {
        return this.apply(_self, args);
      });
    };
    _create = function (namespace) {// 创建命名空间
      var namespace = namespace || _default;
      var cache = {},
        offlineStack = [], // 离线事件
        ret = {
          listen: function (key, fn, last) {
            _listen(key, fn, cache);
            if (offlineStack === null) {
              return;
            }
            if (last === 'last') {
              offlineStack.length && offlineStack.pop()();
            } else {
              each(offlineStack, function () {
                this();
              });
            }
            offlineStack = null;
          },
          one: function (key, fn, last) {
            _remove(key, cache);
            this.listen(key, fn, last);
          },
          remove: function (key, fn) {
            _remove(key, cache, fn);
          },
          trigger: function () {
            var fn,
              args,
              _self = this;
            _unshift.call(arguments, cache);
            args = arguments;
            fn = function () {
              return _trigger.apply(_self, args);
            };
            if (offlineStack) {
              return offlineStack.push(fn);
            }
            return fn();
          }
        };
      return namespace ?
        (namespaceCache[namespace] ? namespaceCache[namespace] :
          namespaceCache[namespace] = ret)
        : ret;
    };
    return {// 返回新函数
      create: _create,
      one: function (key, fn, last) {
        var event = this.create();
        event.one(key, fn, last);
      },
      remove: function (key, fn) {
        var event = this.create();
        event.remove(key, fn);
      },
      listen: function (key, fn, last) {
        var event = this.create();
        event.listen(key, fn, last);
      },
      trigger: function () {
        var event = this.create();
        event.trigger.apply(this, arguments);
      }
    };
  }();
  return Event;
})();

```

## 优缺点

- 优点
  - 时间上的解耦
  - 对象间解耦
  - 符合JavaScript事件驱动的特性

- 缺点
  - 时间及内存的浪费，订阅的消息未必会发生，一直驻留在内存中
  - 对象间的联系深埋背后，导致程序难于维护和理解

总结，刀能切菜也能伤人，凡事有利弊，要合理使用。
